87  // Takes a list of pairs (i,j) and tries to find a solution where
88  // x[i] != x[j]. The function is assumed to be symmetric, i.e.
89  // (1,1) is never a unique optimal solution.

91  template<typename real>
92  void resolve_different(Graph<real,real,real>& graph, std::vector<char>& x, const std::vector<std::pair<int,int> >& pairs)
93  {
94  real infinity = 100; //Does not have to be big
95 
96  // Fix the first element of the pair
97  for (auto itr=pairs.begin(); itr != pairs.end(); ++itr) {
98  int i = itr->first;
99  int j = itr->second;
100 
101  // Extract solution
102  x.at(i) = graph.what_segment(i, Graph<real,real,real>::SOURCE);
103  x.at(j) = graph.what_segment(j, Graph<real,real,real>::SOURCE);
104 
105  //std::cout << i << ',' << j << " : " << int(x[i]) << ',' << int(x[j]) << " ";
106 
107  //ASSERT(x[i]==0 || x[j]==0); // Should not happen, but might due to rounding errors
108 
109  // If x[i]==0 and y[i]==0, we don't know whether there is another
110  // solution where the two are different
111  if (x[i]==0 && x[j]==0) {
112 
113  // Is there also an optimal solution for which x[i]==1?
114  if (graph.what_segment(i, Graph<real,real,real>::SINK) == 1) {
115  // Then fix x[i] to 1
116  graph.add_tweights(i, 0, infinity);
117  graph.mark_node(i);
118  // Resolve the graph
119  graph.maxflow(true);
120  // Extract solution; different if possible
121  x[i] = graph.what_segment(i, Graph<real,real,real>::SOURCE);
122  x[j] = graph.what_segment(j, Graph<real,real,real>::SOURCE);
123  ASSERT(x[i] == 1);
124 
125  if (x[j] == 0) {
126  // We found a solution where x[i]==1 and x[j]==0
127  // std::cout << "A\n";
128  continue;
129  }
130  }
131 
132  // Now fix x[i] to 0
133  graph.add_tweights(i, 2*infinity, 0);
134  graph.mark_node(i);
135  // Resolve the graph
136  graph.maxflow(true);
137  // Does there then exist a solution for which x[j]==1?
138  if (graph.what_segment(j, Graph<real,real,real>::SINK) == 1) {
139  // Then use that solution
140  graph.add_tweights(j, 0, infinity);
141  graph.mark_node(j);
142  // Resolve the graph
143  graph.maxflow(true);
144  x[i] = graph.what_segment(i, Graph<real,real,real>::SOURCE);
145  x[j] = graph.what_segment(j, Graph<real,real,real>::SOURCE);
146  ASSERT(x[i]==0 && x[j]==1);
147  // We found a solution where x[i]==0 and x[j]==1
148  // std::cout << "B\n";
149  continue;
150  }
151 
152  // Otherwise, we only have the solution where x[i]==0 and x[j]==0
153  // std::cout << "C\n";
154  x[i] = 0;
155  x[j] = 0;
156  }
157  else {
158  // std::cout << '\n';
159  }
160  }

309  void AddHigherTerm(int nvars_add, int ind[], real E[])
310  {
311  ASSERT(nvars_add==3);
312 
313  real& E000 = E[0];
314  real& E001 = E[1];
315  real& E010 = E[2];
316  real& E011 = E[3];
317  real& E100 = E[4];
318  real& E101 = E[5];
319  real& E110 = E[6];
320  real& E111 = E[7];
321 
322  int& i = ind[0];
323  int& j = ind[1];
324  int& k = ind[2];
325 
326  // Update number of variables
327  if (i+1>n) n=i+1;
328  if (j+1>n) n=j+1;
329  if (k+1>n) n=k+1;
330 
331  terms0 += E000;
332 
333  real ai = E100 - E000;
334  real aj = E010 - E000;
335  real ak = E001 - E000;
336 
337  real aij = E110 - ai - aj - E000;
338  real aik = E101 - ai - ak - E000;
339  real ajk = E011 - aj - ak - E000;
340 
341  real aijk = E111 - aij - aik - ajk - ai - aj - ak - E000;
342 
343  terms1[i] += ai;
344  terms1[j] += aj;
345  terms1[k] += ak;
346  terms2[unordered_pair(i,j)] += aij;
347  terms2[unordered_pair(i,k)] += aik;
348  terms2[unordered_pair(j,k)] += ajk;
349 
350  if (aijk != 0) {
351  int l = new_vars++;
352 
353  if (aijk < 0) {
354  // Use reduction
355  // -xi*xj*xk = min{z} z*(2-xi-xj-xk)
356  terms2_new[ std::make_pair(i,l)] += aijk;
357  terms2_new[ std::make_pair(j,l)] += aijk;
358  terms2_new[ std::make_pair(k,l)] += aijk;
359  terms1_new[ l ] += -2*aijk;
360  }
361  else if (aijk > 0) {
362  // Use reduction
363  // xi*xj*xk = min{z} z*(1-xi-xj-xk) + xi*xj + xi*xk + xj*xk
364  terms2_new[ std::make_pair(i,l)] += -aijk;
365  terms2_new[ std::make_pair(j,l)] += -aijk;
366  terms2_new[ std::make_pair(k,l)] += -aijk;
367  terms1_new[l] += aijk;
368  terms2[ unordered_pair(i,j)] += aijk;
369  terms2[ unordered_pair(i,k)] += aijk;
370  terms2[ unordered_pair(j,k)] += aijk;
371  }
372  }

376  real minimize()
377  {
378  using namespace std;
379 
380  ASSERT(!graph);
381  int n_nodes = n + new_vars;
382  graph = new Graph<real,real,real>(n_nodes,int(terms2.size()),error_function);
383  graph->add_node(n_nodes);
384 
385  // Add degree-1 terms
386  for (int i=0; i<n; ++i) {
387  real c = terms1[i];
388 
389  // c*x(i)
390  if (c >= 0) {
391  graph->add_tweights(i, c, 0);
392  }
393  else {
394  terms0 += c;
395  graph->add_tweights(i, 0, -c);
396  }
397  }
398  // Add degree-1 terms for new variables
399  for (int ind=0; ind<new_vars; ++ind) {
400  real c = terms1_new[ind];
401  int i = n + ind;
402  if (c >= 0) {
403  graph->add_tweights(i, c, 0);
404  }
405  else {
406  terms0 += c;
407  graph->add_tweights(i, 0, -c);
408  }
409  }
410 
411  // Add degree-2 terms
412  for (auto itr = terms2.begin(); itr != terms2.end(); ++itr) {
413  int i,j;
414  std::tie(i,j) = itr->first;
415  real coef = submodular_coef(itr->second);
416 
417  // Add to graph
418  terms0 += coef;
419  graph->add_tweights(j, 0, -coef);
420  graph->add_edge(i,j, -coef, 0);
421  }
422  // Add degree-2 terms for new variables
423  for (auto itr = terms2_new.begin(); itr != terms2_new.end(); ++itr) {
424  int i,ind;
425  std::tie(i,ind) = itr->first;
426  real coef = submodular_coef(itr->second);
427  int j = n+ind;
428 
429  // Add to graph
430  terms0 += coef;
431  graph->add_tweights(j, 0, -coef);
432  graph->add_edge(i,j, -coef, 0);
433  }
434 
435  // Compute maximum flow
436  real energy = graph->maxflow();
437 
438  // Extract solution
439  x.resize(n);
440  for (int i=0; i<n; ++i) {
441  x[i] = graph->what_segment(i, Graph<real,real,real>::SOURCE);
442  }
443 
444  return terms0 + energy;
445  }
