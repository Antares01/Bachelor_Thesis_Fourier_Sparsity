00100 
00101                         // Extract solution
00102                         x.at(i) = graph.what_segment(i, Graph<real,real,real>::SOURCE);
00103                         x.at(j) = graph.what_segment(j, Graph<real,real,real>::SOURCE);
00104 
00105                         //std::cout << i << ',' << j << " : " << int(x[i]) << ',' << int(x[j]) << "  ";
00106 
00107                         //ASSERT(x[i]==0 || x[j]==0); // Should not happen, but might due to rounding errors
00108 
00109                         // If x[i]==0 and y[i]==0, we don't know whether there is another 
00110                         // solution where the two are different
00111                         if (x[i]==0 && x[j]==0) {
00112 
00113                                 // Is there also an optimal solution for which x[i]==1?
00114                                 if (graph.what_segment(i, Graph<real,real,real>::SINK) == 1) {
00115                                         // Then fix x[i] to 1
00116                                         graph.add_tweights(i, 0, infinity);
00117                                         graph.mark_node(i);
00118                                         // Resolve the graph
00119                                         graph.maxflow(true);
00120                                         // Extract solution; different if possible
00121                                         x[i] = graph.what_segment(i, Graph<real,real,real>::SOURCE);
00122                                         x[j] = graph.what_segment(j, Graph<real,real,real>::SOURCE);
00123                                         ASSERT(x[i] == 1);
00124 
00125                                         if (x[j] == 0) {
00126                                                 // We found a solution where x[i]==1 and x[j]==0
00127                                                 // std::cout << "A\n";
00128                                                 continue;
00129                                         }
00130                                 }
00131 
00132                                 // Now fix x[i] to 0
00133                                 graph.add_tweights(i, 2*infinity, 0);
00134                                 graph.mark_node(i);
00135                                 // Resolve the graph
00136                                 graph.maxflow(true);
00137                                 // Does there then exist a solution for which x[j]==1?
00138                                 if (graph.what_segment(j, Graph<real,real,real>::SINK) == 1) {
00139                                         // Then use that solution
00140                                         graph.add_tweights(j, 0, infinity);
00141                                         graph.mark_node(j);
00142                                         // Resolve the graph
00143                                         graph.maxflow(true);
00144                                         x[i] = graph.what_segment(i, Graph<real,real,real>::SOURCE);
00145                                         x[j] = graph.what_segment(j, Graph<real,real,real>::SOURCE);
00146                                         ASSERT(x[i]==0 && x[j]==1);
00147                                         // We found a solution where x[i]==0 and x[j]==1
00148                                         // std::cout << "B\n";
00149                                         continue;
00150                                 }
00151 
00152                                 // Otherwise, we only have the solution where x[i]==0 and x[j]==0
00153                                 // std::cout << "C\n";
00154                                 x[i] = 0;
00155                                 x[j] = 0;
00156                         }

00158                                 // std::cout << '\n';
